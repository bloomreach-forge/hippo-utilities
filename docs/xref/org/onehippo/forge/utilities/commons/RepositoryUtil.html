94      public static Node ensureParent(final Node parent, final String relFolderPath, final String folderType,
95                                      final Map<String, Object> properties) throws RepositoryException {
96          final List<String> folders = folderParts(relFolderPath);
97          Node tmp = parent;
98          for (String folder : folders) {
99              tmp = createFolder(tmp, folderType, folder, properties);
100         }
101         return tmp;
102     }

104     /**
105      * For given path, create a list of separate node names.
106      *
107      * @param folderPath repository folder path, like {@code  /foo/bar/path/}
108      * @return folder names, like {@code foo,bar,path}
109      */
110     public static List<String> folderParts(final String folderPath) {
111         if (folderPath.indexOf('/') == -1) {
112             return Collections.emptyList();
113         }
114         final List<String> folders = new ArrayList<String>();
115         final String[] parts = folderPath.split("/");
116         for (String part : parts) {
117             if (part.trim().length() != 0) {
118                 folders.add(part);
119             }
120         }
121         return folders;
122     }

124     public static Node createFolder(Node parent, final String type, final String folderName,
125                                     Map<String, Object> properties) throws RepositoryException {
126         if (parent.hasNode(folderName)) {
127             return parent.getNode(folderName);
128         }
129 
130         final Node node = parent.addNode(folderName, type);
131         for (Map.Entry<String, Object> entry : properties.entrySet()) {
132             Object obj = entry.getValue();
133             if (obj instanceof String[]) {
134                 node.setProperty(entry.getKey(), (String[]) obj);
135             } else if (obj instanceof String) {
136                 node.setProperty(entry.getKey(), (String) obj);
137             }
138             // TODO add more types here
139         }
140 
141         node.addMixin("hippo:harddocument");
142         parent.save();
143 
144         return node;
145     }

147     public static Node createHardHandle(final Node parent, final String name, final boolean overwrite)
148             throws RepositoryException {
149         if (parent.hasNode(name)) {
150             if (overwrite) {
151                 log.info("Removing existing binary document: " + name);
152                 parent.getNode(name).remove();
153             } else {
154                 log.info("Binary already exists, skipping: " + name);
155                 return parent.getNode(name);
156             }
157         }
158         // add handle
159         Node handle = parent.addNode(name, "hippo:handle");
160         handle.addMixin("hippo:hardhandle");
161         return handle;
162     }

164     public static Node createDocumentNode(final Node parent, final String name, final String type)
165             throws RepositoryException {
166         // add handle
167         final Node handle = parent.addNode(name, "hippo:handle");
168         handle.addMixin("hippo:hardhandle");
169         // add document node
170         final Node document = handle.addNode(name, type);
171         document.addMixin("hippo:harddocument");
172         return document;
173     }

188     public static Node getDocumentVariant(final Node handle, final String primaryType, final String hippoStdState) throws RepositoryException {
189         if (handle == null || !handle.isNodeType("hippo:handle")) {
190             return null;
191         }
192         final NodeIterator nodeIterator = handle.getNodes();
193         while (nodeIterator.hasNext()) {
194             final Node childNode = nodeIterator.nextNode();
195             if (childNode != null && childNode.isNodeType(primaryType)) {
196                 if (hippoStdState == null) {
197                     return childNode;
198                 }
199                 final Property state = childNode.getProperty("hippostd:state");
200                 if (state != null && hippoStdState.equals(state.getString())) {
201                     return childNode;
202                 }
203             }
204         }
205         return null;
206     }
