57      @SuppressWarnings(value = "unchecked")
58      public static <T> T getValue(final Value value, final Class<T> clazz) throws RepositoryException {
59          final int valueType = value.getType();
60          if (int.class.equals(clazz) || Integer.class.equals(clazz)) {
61              throw new UnsupportedOperationException("JCR doesn't support integer values. Please use the type Long to get the value.");
62          }
63          switch (valueType) {
64              case PropertyType.BOOLEAN:
65                  if (Boolean.class.equals(clazz) || boolean.class.equals(clazz)) {
66                      return (T) (Boolean) value.getBoolean();
67                  }
68                  break;
69              case PropertyType.DATE:
70                  if (Calendar.class.equals(clazz)) {
71                      return (T) value.getDate();
72                  }
73                  break;
74              case PropertyType.DOUBLE:
75                  if (Double.class.equals(clazz) || double.class.equals(clazz)) {
76                      return (T) (Double) value.getDouble();
77                  }
78                  break;
79              case PropertyType.LONG:
80                  if (Long.class.equals(clazz) || long.class.equals(clazz)) {
81                      return (T) (Long) value.getLong();
82                  }
83                  break;
84              case PropertyType.STRING:
85                  if (String.class.equals(clazz)) {
86                      return (T) value.getString();
87                  }
88                  break;
89          }
90          throw new UnsupportedOperationException("The type "+clazz+" with value type '"+valueType+"' is not supported.");
91      }

93      @SuppressWarnings(value = "unchecked")
94      public static <T> Collection<T> getPropertyValues(final Property property, final Class<? extends Collection> clazz, final Class<T> genericType) throws RepositoryException, IllegalAccessException, InstantiationException {
95          if (!property.getDefinition().isMultiple()) {
96              return Collections.emptyList();
97          }
98          final Collection<T> collection = clazz.newInstance();
99          for (final Value value : property.getValues()) {
100             collection.add(getValue(value, genericType));
101         }
102         return collection;
103     }

106     public static <T> Collection<T> getPropertyValues(final Node node, final String path, final Class<? extends Collection> clazz, final Class<T> genericType) throws RepositoryException, InstantiationException, IllegalAccessException {
107         if (!node.hasProperty(path)) {

113     public static <T> T getPropertyValue(final Property property, final Class<T> clazz) throws RepositoryException {
114         if (property.getDefinition().isMultiple() || clazz.isArray()) {
115             throw new UnsupportedOperationException("The method get property value can only be used for non array types. Use getPropertyValues() instead.");
116         }
117         return getValue(property.getValue(), clazz);
118     }

127     public static void setProperty(final Node node, final String relativePath, final Object object) throws RepositoryException {
128         logger.debug("Setting property '{}' to value '{}'", new Object[] {node.getPath()+"/"+relativePath, object});
129         if (!node.isCheckedOut()) {
130             node.checkout();
131         }
132         if (object == null) {
133             if (node.hasProperty(relativePath)) {
134                 node.getProperty(relativePath).remove();
135             }
136         } else if (object instanceof String) {
137             node.setProperty(relativePath, (String) object);
138         } else if (object instanceof String[]) {
139             node.setProperty(relativePath, (String[]) object);
140         } else if (object instanceof Boolean) {
141             node.setProperty(relativePath, (Boolean) object);
142         } else if (object instanceof Double) {
143             node.setProperty(relativePath, (Double) object);
144         } else if (object instanceof Integer) {
145             node.setProperty(relativePath, (Integer) object);
146         } else if (object instanceof Long) {
147             node.setProperty(relativePath, (Long) object);
148         } else if (object instanceof Calendar) {
149             node.setProperty(relativePath, (Calendar)object);
150         } else {
151             throw new UnsupportedOperationException("The method setProperty for object " + object + " is not supported yet.");
152         }
153     }

155     public static <T> void setPropertyValues(final Property property, final Collection<T> values) throws RepositoryException {
156         final Value[] newValues = new Value[values.size()];
157         int index = 0;
158         for (T object : values) {
159             newValues[index++] = createValue(object);
160         }
161         property.setValue(newValues);
162     }

164     public static <T> void setPropertyValues(final Node node, final String relativePath, final Collection<T> values) throws RepositoryException {
165         logger.debug("Setting property '{}' values from collection.", node.getPath()+"/"+relativePath);
166         if (!node.isCheckedOut()) {
167             node.checkout();
168         }
169         if (values == null) {
170             if (node.hasProperty(relativePath)) {
171                 logger.debug("Remove property "+node.getPath()+"/"+relativePath);
172                 node.getProperty(relativePath).remove();
173             }
174         } else {
175             ArrayList<Value> newValues = new ArrayList<Value>();
176             for (T object : values) {
177                 newValues.add(createValue(object));
178             }
179             node.setProperty(relativePath, newValues.toArray(new Value[newValues.size()]));
180         }
181     }

183     public static Value createValue(Object object) {
184         if (object instanceof String) {
185             return new StringValue((String) object);
186         } else if (object instanceof Long) {
187             return new LongValue((Long) object);
188         } else if (object instanceof Double) {
189             return new DoubleValue((Double) object);
190         } else if (object instanceof Calendar) {
191             return new DateValue((Calendar) object);
192         } else if (object instanceof Boolean) {
193             return new BooleanValue((Boolean) object);
194         }
195         throw new UnsupportedOperationException("Create value doesn't support the type '" + object.getClass() + "'");
196     }

198     public static List<Class<?>> getActualTypeParameters(Field field) {
199         final List<Class<?>> actualTypeParameters = new ArrayList<Class<?>>();
200         final Type genericFieldType = field.getGenericType();
201         if (genericFieldType instanceof ParameterizedType) {
202             final ParameterizedType type = (ParameterizedType) genericFieldType;
203             Type[] fieldArgTypes = type.getActualTypeArguments();
204             for (Type fieldArgType : fieldArgTypes) {
205                 actualTypeParameters.add((Class) fieldArgType);
206             }
207         }
208         return actualTypeParameters;
209     }

211     public static <T extends Annotation> T getGenericAnnotation(final Field field, final int parameterIndex, final Class<T> annotationClazz) {
212         final List<Class<?>> actualTypeParameters = getActualTypeParameters(field);
213         if (actualTypeParameters.size() <= parameterIndex) {
214             return null;
215         }
216         return actualTypeParameters.get(parameterIndex).getAnnotation(annotationClazz);
217     }
