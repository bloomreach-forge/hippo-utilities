57      public SimpleFtpClient(final String userName, final String password, final String server, final int port) {
58          this.userName = userName;
59          this.password = password;
60          this.server = server;
61          this.port = port;

126     /**
127      * Save given stream into given directory
128      *
129      * @param stream            stream we want to save
130      * @param fileName          name of the file
131      * @param directory         directory file should be saved to.
132      *                          Note: directory is relative to user home directory, even when provided as absolute directory (e.g. /foo/bar is treated as foo/bar)
133      * @param binaryUpload      is this a binary upload
134      * @param overwrite         overwrite existing file?
135      * @param createDirectories create directories if not exist
136      * @return SimpleFtpClientResult.CREATED if file created, SimpleFtpClientResult.FILE_OVERWRITTEN; if file overwritten, SimpleFtpClientResult.ERROR otherwise;
137      */
138     public SimpleFtpClientResult saveFile(final InputStream stream, final String fileName, final String directory, final boolean binaryUpload, final boolean overwrite, final boolean createDirectories) {
139 
140         try {
141             connectClient();
142 
143             if (binaryUpload) {
144                 client.setFileType(FTP.BINARY_FILE_TYPE);
145             }
146             if (createDirectories) {
147                 SimpleFtpClientResult result = createDirectories(directory);
148                 if (result != SimpleFtpClientResult.CREATED) {
149                     return SimpleFtpClientResult.FAILED_CREATE_DIR;
150                 }
151             }
152             changeToDir(directory);
153 
154             if (overwrite) {
155                 client.storeFile(fileName, stream);
156                 stream.close();
157                 return SimpleFtpClientResult.FILE_OVERWRITTEN;
158             } else {
159                 // check if file exists:
160                 String[] names = client.listNames();
161                 boolean stored = false;
162                 for (String name : names) {
163                     if (fileName.equals(name)) {
164                         client.storeUniqueFile(fileName, stream);
165                         stored = true;
166                         break;
167                     }
168                 }
169                 if (!stored) {
170                     client.storeFile(fileName, stream);
171                 }
172 
173                 stream.close();
174                 return SimpleFtpClientResult.CREATED;
175             }
176 
177 
178         } catch (IOException e) {
179             log.error("Error saving file to FTP server: {}", server, e);
180             return SimpleFtpClientResult.ERROR;
181         }
182 
183     }

185     /**
186      * Return input stream for given file name
187      *
188      * @param fileName  name of the remote file
189      * @param directory remote directory
190      * @return input stream or null
191      */
192     public InputStream getInputStream(final String fileName, final String directory) {
193 
194         try {
195             connectClient();
196             changeToDir(directory);
197             return client.retrieveFileStream(fileName);
198 
199         } catch (IOException e) {
200             log.error("Error retrieving file from FTP server: {}", server, e);
201         }
202         return null;
203     }

205     public String getString(final String fileName, final String directory) {
206 
207         InputStream inputStream = null;
208         try {
209             connectClient();
210             changeToDir(directory);
211             inputStream = client.retrieveFileStream(fileName);
212             String result = IOUtils.toString(inputStream, Charset.defaultCharset());
213             client.completePendingCommand();
214             return result;
215 
216         } catch (IOException e) {
217             log.error("Error retrieving file from FTP server: {}", server, e);
218         } finally {
219             if (inputStream != null) {
220                 try {
221                     inputStream.close();
222                 } catch (IOException e) {
223                     log.error("Error closing stream", e);
224                 }
225 
226             }
227         }
228         return null;
229     }

231     /**
232      * Changes working director
233      *
234      * @param directory directory (relative to user home dir)
235      * @throws IOException on error
236      */
237     private void changeToDir(final String directory) throws IOException {
238         Iterable<String> iterable = Splitter.on('/').omitEmptyStrings().split(directory);
239         // note: important: we first go to user home
240         client.changeWorkingDirectory(getHomeDirectory());
241         for (String dir : iterable) {
242             client.changeWorkingDirectory(dir);
243         }
244     }
245 
246     /**
247      * Connects client if not connected yet. must be called before we do any actions
248      *
249      * @throws IOException on error
250      */
251     public final void connectClient() throws IOException {
252         if (!client.isConnected()) {
253             client.connect(server, port);
254             client.login(userName, password);
255             int reply = client.getReplyCode();
256             if (!FTPReply.isPositiveCompletion(reply)) {
257                 client.disconnect();
258                 log.error("Negative reply form FTP server, aborting, id was {}:", reply);
259                 throw new IOException("failed to connect to FTP server");
260             }
261             // initialize working directory
262             homeDirectory = client.printWorkingDirectory();
263         }
264     }

267     /**
268      * Creates directories on FTP server
269      *
270      * @param directory directory to create
271      * @return SimpleFtpClientResult.CREATED on succes SimpleFtpClientResult.ERROR on error
272      * @throws IOException on error
273      */
274     private SimpleFtpClientResult createDirectories(final String directory) throws IOException {
275         Iterable<String> iterable = Splitter.on('/').omitEmptyStrings().split(directory);
276         for (String dir : iterable) {
277             boolean dirExists = client.changeWorkingDirectory(dir);
278             if (!dirExists) {
279                 // try to create directory:
280                 client.makeDirectory(dir);
281                 dirExists = client.changeWorkingDirectory(dir);
282             }
283             if (!dirExists) {
284                 log.error("failed to change FTP directory (forms), not doing anything");
285                 return SimpleFtpClientResult.ERROR;
286             }
287         }
288         return SimpleFtpClientResult.CREATED;
289     }

317     public String getHomeDirectory() {
318         if (homeDirectory == null) {
319             try {
320                 connectClient();
321             } catch (IOException e) {
322                 log.error("Error getting home dir:", e);
323             }
324         }
325         return homeDirectory;
326     }
