16  package org.onehippo.forge.utilities.repository.scheduling;
17  
18  import javax.jcr.Node;
19  import javax.jcr.RepositoryException;
20  import javax.jcr.Session;
21  
22  import org.hippoecm.repository.api.HippoNodeType;
23  import org.hippoecm.repository.util.JcrUtils;
24  import org.onehippo.repository.scheduling.RepositoryJob;
25  import org.onehippo.repository.scheduling.RepositoryJobExecutionContext;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  /**
30   * A Generic Updater Executing Scheduler Job implementation.
31   * <P>
32   * Configuration example (assuming that you have an updater at <CODE>/hippo:configuration/hippo:update/hippo:registry/DemoUpdater</CODE>):
33   * </P>
34   * <PRE>
35   * /hippo:configuration/hippo:modules/scheduler/hippo:moduleconfig/system/demoUpdaterExecutor:
36   *   jcr:primaryType: hipposched:repositoryjob
37   *   hipposched:attributeNames: [updatername]
38   *   hipposched:attributeValues: [DemoUpdater]
39   *   hipposched:enabled: true
40   *   hipposched:repositoryJobClass: org.onehippo.forge.utilities.repository.scheduling.UpdaterExecutingJob
41   *   /hipposched:triggers:
42   *     jcr:primaryType: hipposched:triggers
43   *     /nightly:
44   *       jcr:primaryType: hipposched:crontrigger
45   *       jcr:mixinTypes: ['hippo:lockable', 'mix:lockable']
46   *       hipposched:cronExpression: 0 0 3 * * ?
47   *       hipposched:enabled: true
48   * </PRE>
49   */
50  public class UpdaterExecutingJob implements RepositoryJob {
51  
52      private static final Logger log = LoggerFactory.getLogger(UpdaterExecutingJob.class);
53  
54      private static final String CONFIG_UPDATER_NAME = "updatername";
55  
56      private static final String UPDATE_PATH = "/hippo:configuration/hippo:update";
57      private static final String UPDATE_QUEUE_PATH = UPDATE_PATH + "/hippo:queue";
58      private static final String UPDATE_REGISTRY_PATH = UPDATE_PATH + "/hippo:registry";
59  
60      @Override
61      public void execute(final RepositoryJobExecutionContext context) throws RepositoryException {
62          String updaterName = context.getAttribute(CONFIG_UPDATER_NAME);
63  
64          if (updaterName != null) {
65              updaterName = updaterName.trim();
66          }
67  
68          if (updaterName == null || updaterName.isEmpty()) {
69              log.warn("Skipping updater executing job due to the blank updater name configuration.");
70              return;
71          }
72  
73          Session session = null;
74  
75          try {
76              session = context.createSystemSession();
77  
78              final Node queueNode = session.getNode(UPDATE_QUEUE_PATH);
79              final Node registryNode = session.getNode(UPDATE_REGISTRY_PATH);
80  
81              if (!registryNode.hasNode(updaterName)) {
82                  log.error("Skipping updater script execution because the configured updater node doesn't exist at {}/{}.",
83                          UPDATE_REGISTRY_PATH, updaterName);
84                  return;
85              }
86  
87              final Node registeredUpdaterNode = registryNode.getNode(updaterName);
88              final Node queuedUpdaterNode = JcrUtils.copy(registeredUpdaterNode, registeredUpdaterNode.getName(),
89                      queueNode);
90              queuedUpdaterNode.setProperty(HippoNodeType.HIPPOSYS_STARTEDBY, session.getUserID());
91              session.save();
92          } finally {
93              if (session != null) {
94                  session.logout();
95              }
96          }
97      }
98  
99  }
