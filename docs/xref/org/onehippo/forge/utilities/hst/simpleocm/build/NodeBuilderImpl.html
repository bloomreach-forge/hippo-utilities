48      /**
49       * Builds a property by routing to helper method
50       *
51       * @param node                 the node to build the property / child node
52       * @param relPath              the relative path for the new property / child node
53       * @param value                the value to build the property / child node of
54       * @param type                 the type of the the field
55       * @param actualTypeParameters the type parameters of the field type
56       * @throws RepositoryException         if building fails
57       * @throws ContentNodeBindingException if building fails
58       */
59      @SuppressWarnings("unchecked")
60      private void buildProperty(final Node node, final String relPath, final Object value, final Class<?> type, final List<Class<?>> actualTypeParameters) throws RepositoryException, ContentNodeBindingException {
61          final JcrNodeType nodeTypeAnnotation = type.getAnnotation(JcrNodeType.class);
62          Class<?> firstTypeParameter = getListItem(actualTypeParameters, 0);
63          if (nodeTypeAnnotation != null) {
64              build(node, relPath, value, nodeTypeAnnotation.sameNameSiblings());
65          } else if ("*".equals(relPath) && Map.class.equals(type)) {
66              buildChildNodesFromMap(node, (Map) value, actualTypeParameters, firstTypeParameter);
67          } else if ((Set.class.equals(type) || List.class.equals(type)) && firstTypeParameter != null && firstTypeParameter.getAnnotation(JcrNodeType.class) != null) {
68              buildChildNodesFromCollection(node, relPath, (Collection) value);
69          } else if (Set.class.equals(type) || List.class.equals(type)) {
70              GenericsUtil.setPropertyValues(node, relPath, (Collection) value);
71          } else {
72              GenericsUtil.setProperty(node, relPath, value);
73          }
74      }

76      /**
77       * Builds the child nodes from a collection, removes all existing child nodes with that name. An update behavior is
78       * with a collection not possible, because it's not possible to make associations between nodes and collection
79       * members. So the annotation value sameNameSiblings in JcrNodeType is ignored.
80       *
81       * @param parent     the parent node to append the new child nodes to
82       * @param relPath    the relative path of the new child nodes (sameNameSiblings should be true @see
83       *                   rg.onehippo.forge.utilities.hst.simpleocm.JcrNodeType)
84       * @param collection the collection
85       * @throws ContentNodeBindingException   if building the child nodes fails
86       * @throws javax.jcr.RepositoryException if removing old child nodes with fails
87       */
88      private void buildChildNodesFromCollection(final Node parent, final String relPath, final Collection collection) throws ContentNodeBindingException, RepositoryException {
89          while (parent.hasNode(relPath)) {
90              final Node childNode = parent.getNode(relPath);
91              logger.debug("Removing existing child node '{}'", childNode.getPath());
92              childNode.remove();
93          }
94          if (collection == null) {
95              return;
96          }
97          logger.debug("Building child nodes from collection");
98          for (Object object : collection) {
99              build(parent, relPath, object, true);
100         }
101     }

103     /**
104      * Builds child nodes from a map, the key is providing the path for the child node
105      *
106      * @param parent               the parent node to add the child nodes to
107      * @param map                  the map to build child nodes of
108      * @param actualTypeParameters the generic type parameters of the map
109      * @param firstTypeParameter   the first type parameter
110      * @throws ContentNodeBindingException   if building the child nodes fails
111      * @throws javax.jcr.RepositoryException if removing a existing child node with clashing name - key fails
112      */
113     private void buildChildNodesFromMap(final Node parent, final Map map, final List<Class<?>> actualTypeParameters, final Class<?> firstTypeParameter) throws ContentNodeBindingException, RepositoryException {
114         if (actualTypeParameters.size() != 2 || !String.class.equals(firstTypeParameter)) {
115             throw new UnsupportedOperationException("The path * is only supported for building Map<String, Object@JcrNodeType>.");
116         }
117         final JcrNodeType typeParameterAnnotation = actualTypeParameters.get(1).getAnnotation(JcrNodeType.class);
118         if (typeParameterAnnotation == null) {
119             throw new UnsupportedOperationException("The path * is only supported for building Map<String, Object@JcrNodeType>.");
120         }
121 
122         removeMissingChildren(parent, map, typeParameterAnnotation);
123 
124         if (map == null) {
125             return;
126         }
127         logger.debug("Building child nodes from map");
128         for (Object object : map.entrySet()) {
129             if (object instanceof Map.Entry) {
130                 final Map.Entry entry = (Map.Entry)object;
131                 build(parent, String.valueOf(entry.getKey()), entry.getValue(), typeParameterAnnotation.sameNameSiblings());
132             }
133         }
134     }
135 
136     private void removeMissingChildren(final Node parent, final Map map, final JcrNodeType typeParameterAnnotation) throws RepositoryException {
137         final NodeIterator childNodeIterator = parent.getNodes();
138         while (childNodeIterator.hasNext()) {
139             final Node childNode = childNodeIterator.nextNode();
140             final String decodedChildNodeName = NodeNameCodec.decode(childNode.getName());
141             if (!typeParameterAnnotation.value().equals(childNode.getPrimaryNodeType().getName())) {
142                 continue;
143             }
144             if (map == null) {
145                 if (logger.isDebugEnabled()) {
146                     logger.debug("Map is null removing '{}' of type {}", childNode.getPath(), typeParameterAnnotation.value());
147                 }
148                 childNodeIterator.remove();
149             } else if (!map.containsKey(decodedChildNodeName)) {
150                 if (logger.isDebugEnabled()) {
151                     logger.debug("Map doesn't contain child node of name {}, removing '{}'", decodedChildNodeName, childNode.getPath());
152                 }
153                 childNodeIterator.remove();
154             }
155         }
156     }

158     /**
159      * Build property
160      *
161      * @param node    the node to set / append the property / child node
162      * @param relPath the relative path for the new property / child node
163      * @param field   the field to get the data / information from to build
164      * @param obj     the object the field is bound to
165      * @throws ContentNodeBindingException
166      */
167     public void buildProperty(Node node, String relPath, Field field, Object obj) throws ContentNodeBindingException {
168         try {
169             if (!field.isAccessible()) {
170                 field.setAccessible(true);
171             }
172             final List<Class<?>> actualTypeParameters = GenericsUtil.getActualTypeParameters(field);
173             final Object value = field.get(obj);
174             final Class<?> type = field.getType();
175             buildProperty(node, relPath, value, type, actualTypeParameters);
176         } catch (IllegalAccessException accessException) {
177             throw new ContentNodeBindingException("Error building property for object '" + obj.getClass() + "' relative path '" + relPath + "'", accessException);
178         } catch (RepositoryException repositoryException) {
179             throw new ContentNodeBindingException("Error building property for object '" + obj.getClass() + "' relative path '" + relPath + "'", repositoryException);
180         }
181     }

183     /**
184      * @see NodeBuilder
185      */
186     public Node createNode(final Node parent, final String name, final Object obj, final boolean sameNameSiblings) throws ContentNodeBindingException {
187         final JcrNodeType annotation = obj.getClass().getAnnotation(JcrNodeType.class);
188         if (annotation == null) {
189             return null;
190         }
191         final String nodeName = NodeNameCodec.encode(name);
192         try {
193             if (!parent.isCheckedOut()) {
194                 parent.getSession().getWorkspace().getVersionManager().checkout(parent.getPath());
195             }
196             if (!sameNameSiblings && parent.hasNode(nodeName)) {
197                 return parent.getNode(nodeName);
198             } else {
199                 return parent.addNode(nodeName, annotation.value());
200             }
201         } catch (RepositoryException repositoryException) {
202             throw new ContentNodeBindingException("Error creating node '" + name + "' for class " + obj.getClass(), repositoryException);
203         }
204     }

206     /**
207      * Bind all @see org.onehippo.forge.utilities.hst.simpleocm.JcrPath annotated fields to the passed node. Also takes care of
208      * inheritance and loops up, to check the fields of the super classes.
209      * The session, or node needs to be saved to persist the changes.
210      *
211      * @param node   the node to populate
212      * @param object the @see org.onehippo.forge.utilities.hst.simpleocm.JcrNodeType annotated object
213      * @throws org.hippoecm.hst.content.beans.ContentNodeBindingException
214      *          also @see org.hippoecm.hst.content.beans.ContentNodeBinder#bind
215      */
216     public boolean bind(Object object, Node node) throws ContentNodeBindingException {
217         Class<?> clazz = object.getClass();
218         while (!Object.class.equals(clazz)) {
219             buildPropertiesForClass(node, object, clazz);
220             JcrNodeType nodeTypeAnnotation = clazz.getAnnotation(JcrNodeType.class);
221             if (nodeTypeAnnotation != null && !ContentNodeBinder.class.equals(nodeTypeAnnotation.binder())) {
222                   instantiateBinder(nodeTypeAnnotation.binder()).bind(object, node);
223             }
224             clazz = clazz.getSuperclass();
225         }
226         return true;
227     }

229     /**
230      * Builds the properties and child nodes for a class.
231      *
232      * @param node   the node to populate
233      * @param object the annotated object
234      * @param clazz  the class, to get the annotations from
235      * @throws org.hippoecm.hst.content.beans.ContentNodeBindingException
236      *          if building a property / child node fails
237      */
238     private void buildPropertiesForClass(final Node node, final Object object, final Class<?> clazz) throws ContentNodeBindingException {
239         logger.debug("Building properties for class '{}'", clazz);
240         final Field fields[] = clazz.getDeclaredFields();
241         for (Field field : fields) {
242             final Annotation annotations[] = field.getAnnotations();
243             for (Annotation annotation : annotations) {
244                 if (!(annotation instanceof JcrPath)) {
245                     continue;
246                 }
247                 String relativePath = ((JcrPath) annotation).value();
248                 if (StringUtils.isBlank(relativePath)) {
249                     relativePath = field.getName();
250                 }
251                 final Class<? extends Converter> converterClass = ((JcrPath) annotation).converter();
252                 if (!Converter.class.equals(converterClass)) {
253                     try {
254                         logger.debug("Building property '{}' for field '{}' with converter '{}'", new Object[]{relativePath, field.getName(), converterClass});
255                         converterClass.newInstance().buildProperty(node, relativePath, field, object);
256                     } catch (IllegalAccessException accessException) {
257                         throw new ContentNodeBindingException("Error building the property '" + relativePath + "' for class " + clazz, accessException);
258                     } catch (InstantiationException instantiationException) {
259                         throw new ContentNodeBindingException("Error building the property '" + relativePath + "' for class " + clazz, instantiationException);
260                     }
261                 } else {
262                     logger.debug("Building property '{}' for field '{}'", relativePath, field.getName());
263                     buildProperty(node, relativePath, field, object);
264                 }
265             }
266         }
267     }

269     /**
270      * Builds the node with all annotated properties and child nodes.
271      * The session, or node needs to be saved to persist the changes.
272      * @param parent   parent node to add the child to
273      * @param nodeName the name of the new node to create
274      * @param object   the object to marshal to JCR
275      * @return a new {@link Node} or {@literal null} if the nodetype cannot be annotated
276      * @throws ContentNodeBindingException
277      */
278     public Node build(final Node parent, final String nodeName, final Object object) throws ContentNodeBindingException {
279         final Class<?> clazz = object.getClass();
280         final JcrNodeType nodeTypeAnnotation = clazz.getAnnotation(JcrNodeType.class);
281         if (nodeTypeAnnotation == null) {
282             return null;
283         }
284         try {
285             return build(parent, nodeName, object, nodeTypeAnnotation.sameNameSiblings());
286         } catch (RepositoryException repositoryException) {
287             throw new ContentNodeBindingException("Error building node '"+nodeName+"' from object "+object, repositoryException);
288         }
289     }

291     /**
292      * Builds the node with all annotated properties and child nodes. Local helper to use the sameNameSiblings behavior
293      * internally as needed.
294      *
295      * @param parent   the parent node
296      * @param nodeName the name of the new node (appended to the parent)
297      * @param object   the annotated bean, which will be stored in the repository
298      * @param sameNameSiblings if true will not create the child node, but instead take the existing node with the passed name
299      * @return the build bean
300      * @throws RepositoryException if building the node fails
301      * @throws org.hippoecm.hst.content.beans.ContentNodeBindingException if building the node fails
302      */
303     private Node build(final Node parent, final String nodeName, final Object object, final boolean sameNameSiblings) throws ContentNodeBindingException, RepositoryException {
304         if (object == null) {
305             if (parent.hasNode(nodeName)) {
306                 parent.getNode(nodeName).remove();
307             }
308             return null;
309         }
310         final Class<?> clazz = object.getClass();
311         final JcrNodeType nodeTypeAnnotation = clazz.getAnnotation(JcrNodeType.class);
312         if (nodeTypeAnnotation == null) {
313             return null;
314         }
315         logger.debug("Building child node '{}' for node '{}' from object '{}'", new Object[]{nodeName, parent.getPath(), object});
316         final Node node = createNode(parent, nodeName, object, sameNameSiblings);
317         if (node == null) {
318             return null;
319         }
320         bind(object, node);
321         if (!ContentNodeBinder.class.equals(nodeTypeAnnotation.binder())) {
322             final ContentNodeBinder nodeBinder = instantiateBinder(nodeTypeAnnotation.binder());
323             nodeBinder.bind(object, node);
324         }
325         return node;
326     }

328     /**
329      * Instantiates a builder
330      *
331      * @param nodeBinder the class of the node builder to initialize
332      * @return this if clazz equals this.getClass() otherwise the new instantiated node builder
333      * @throws org.hippoecm.hst.content.beans.ContentNodeBindingException
334      *          if instantiating the builder fails
335      */
336     private ContentNodeBinder instantiateBinder(Class<? extends ContentNodeBinder> nodeBinder) throws ContentNodeBindingException {
337         if (this.getClass().equals(nodeBinder)) {
338             return this;
339         }
340         try {
341             return nodeBinder.newInstance();
342         } catch (IllegalAccessException accessException) {
343             throw new ContentNodeBindingException("Error occurred instantiating the node builder " + nodeBinder, accessException);
344         } catch (InstantiationException instantiationException) {
345             throw new ContentNodeBindingException("Error occurred instantiating the node builder " + nodeBinder, instantiationException);
346         }
347     }

349     private <T> T getListItem(List<T> list, int index) {
350         if (list.size() > index) {
351             return list.get(index);
352         }
353         return null;
354     }
