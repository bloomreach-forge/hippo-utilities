43      @Test
44      public void testSetProperties() throws RepositoryException {
45          Node node = JcrMockUp.mockJcrNode("/content.xml");
46          node.setProperty("string", "foobar");
47          node.setProperty("multistring", new String[]{"foo", "bar"});
48          node.setProperty("boolean", true);
49          node.setProperty("long", 1234);
50          node.setProperty("double", 123.2);
51          Calendar calendar = Calendar.getInstance();
52          node.setProperty("date", calendar);
53  
54          assert "foobar".equals(node.getProperty("string").getString());
55          assert node.getProperty("boolean").getBoolean();
56          assert new Long(1234).equals(node.getProperty("long").getLong());
57          assert 123.2 == node.getProperty("double").getDouble();
58          assert node.getProperty("date").getDate().getTime().equals(calendar.getTime());
59  
60          assert node.getProperty("multistring").getDefinition().isMultiple();
61          assert node.getProperty("multistring").getValues().length == 2;
62          assert "foo".equals(node.getProperty("multistring").getValues()[0].getString());
63      }

311     @Test
312     public void testPropertyIterator() throws RepositoryException {
313         Session session = JcrMockUp.mockJcrSession("/content.xml");
314         Node node = (Node) session.getItem("/14-9-wie-denk-je-wel-dat-je-bent/14-9-wie-denk-je-wel-dat-je-bent");
315 
316         Set<String> propertyNames = new HashSet<String>();
317         propertyNames.add("jcr:primaryType");
318         propertyNames.add("jcr:mixinTypes");
319         propertyNames.add("jcr:uuid");
320         propertyNames.add("jcrmockup:agendadate_from");
321         propertyNames.add("jcrmockup:agendadate_to");
322         propertyNames.add("jcrmockup:documentdate");
323         propertyNames.add("jcrmockup:generate_anchors");
324         propertyNames.add("jcrmockup:introduction");
325         propertyNames.add("jcrmockup:title");
326         propertyNames.add("hippo:availability");
327         propertyNames.add("hippo:paths");
328         propertyNames.add("hippo:related___pathreference");
329         propertyNames.add("hippostd:holder");
330         propertyNames.add("hippostd:state");
331         propertyNames.add("hippostd:stateSummary");
332         propertyNames.add("hippostd:tags");
333         propertyNames.add("hippostdpubwf:createdBy");
334         propertyNames.add("hippostdpubwf:creationDate");
335         propertyNames.add("hippostdpubwf:lastModificationDate");
336         propertyNames.add("hippostdpubwf:lastModifiedBy");
337         propertyNames.add("hippostdpubwf:publicationDate");
338 
339         boolean propertyNotInSet = false;
340         PropertyIterator propertyIterator = node.getProperties();
341         while (propertyIterator.hasNext()) {
342             Property property = propertyIterator.nextProperty();
343             if (!propertyNames.contains(property.getName())) {
344                 propertyNotInSet = true;
345                 break;
346             }
347             propertyNames.remove(property.getName());
348         }
349 
350         assert propertyNames.size() == 0 && !propertyNotInSet;
351     }

396     @Test
397     public void sameNameSiblingPathTest() throws RepositoryException {
398         Session session = JcrMockUp.mockEmptySession();
399         Node rootNode = session.getRootNode();
400 
401         Node firstFooNode = rootNode.addNode("foo", "jcrmock:testnode");
402         Node secondFooNode = rootNode.addNode("foo", "jcrmock:testnode");
403 
404         Node firstBarNode = secondFooNode.addNode("bar", "jcrmock:testnode");
405         Node secondBarNode = secondFooNode.addNode("bar", "jcrmock:testnode");
406 
407         assert "/foo".equals(firstFooNode.getPath());
408         assert "/foo[2]".equals(secondFooNode.getPath());
409         assert "/foo[2]/bar".equals(firstBarNode.getPath());
410         assert "/foo[2]/bar[2]".equals(secondBarNode.getPath());
411     }
