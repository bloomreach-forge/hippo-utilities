85      @Test
86      public void testCollectionUpdate() throws RepositoryException, ContentNodeBindingException {
87          Session session = JcrMockUp.mockJcrSession("/content.xml");
88          // load bean
89          BeanLoader beanLoader = new BeanLoaderImpl();
90          Agenda agenda = new Agenda();
91          Node agendaNode = (Node) session.getItem("/14-9-wie-denk-je-wel-dat-je-bent/14-9-wie-denk-je-wel-dat-je-bent");
92          beanLoader.loadBean(agendaNode, agenda);
93  
94          // change bean
95          HippoHtml html1 = new HippoHtml();
96          html1.content = "content1";
97          agenda.paragraphs.add(html1);
98  
99          // write bean back
100         NodeBuilder nodeBuilder = new NodeBuilderImpl();
101         nodeBuilder.bind(agenda, agendaNode);
102 
103         // add second node and update
104         HippoHtml html2 = new HippoHtml();
105         html2.content = "content2";
106         agenda.paragraphs.add(html2);
107         nodeBuilder.bind(agenda, agendaNode);
108 
109         int containsContent1 = 0;
110         int containsContent2 = 0;
111         NodeIterator nodeIterator = agendaNode.getNodes("jcrmockup:paragraph");
112         while (nodeIterator.hasNext()) {
113             Node paragraphNode = nodeIterator.nextNode();
114             if (html1.content.equals(paragraphNode.getProperty("hippostd:content").getString())) {
115                 containsContent1++;
116             }
117             if (html2.content.equals(paragraphNode.getProperty("hippostd:content").getString())) {
118                 containsContent2++;
119             }
120         }
121         assert containsContent1 == 1 && containsContent2 == 1;
122     }
